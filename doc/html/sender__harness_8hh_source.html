329  void execute(TCPSender &, std::deque<TCPSegment> &segments) const {
330  if (segments.empty()) {
331  throw SegmentExpectationViolation::violated_verb("existed");
332  }
333  TCPSegment seg = std::move(segments.back());
334  segments.pop_back();
335  if (ack.has_value() and seg.header().ack != ack.value()) {
336  throw SegmentExpectationViolation::violated_field("ack", ack.value(), seg.header().ack);
337  }
338  if (rst.has_value() and seg.header().rst != rst.value()) {
339  throw SegmentExpectationViolation::violated_field("rst", rst.value(), seg.header().rst);
340  }
341  if (syn.has_value() and seg.header().syn != syn.value()) {
342  throw SegmentExpectationViolation::violated_field("syn", syn.value(), seg.header().syn);
343  }
344  if (fin.has_value() and seg.header().fin != fin.value()) {
345  throw SegmentExpectationViolation::violated_field("fin", fin.value(), seg.header().fin);
346  }
347  if (seqno.has_value() and seg.header().seqno != seqno.value()) {
348  throw SegmentExpectationViolation::violated_field("seqno", seqno.value(), seg.header().seqno);
349  }
350  if (ackno.has_value() and seg.header().ackno != ackno.value()) {
351  throw SegmentExpectationViolation::violated_field("ackno", ackno.value(), seg.header().ackno);
352  }
353  if (win.has_value() and seg.header().win != win.value()) {
354  throw SegmentExpectationViolation::violated_field("win", win.value(), seg.header().win);
355  }
356  if (payload_size.has_value() and seg.payload().size() != payload_size.value()) {
357  throw SegmentExpectationViolation::violated_field(
358  "payload_size", payload_size.value(), seg.payload().size());
359  }
360  if (seg.length_in_sequence_space() > TCPConfig::MAX_PAYLOAD_SIZE) {
361  throw SegmentExpectationViolation("packet has length_including_flags (" +
362  std::to_string(seg.length_in_sequence_space()) +
363  ") greater than the maximum");
364  }
365  if (data.has_value() and seg.payload().str() != data.value()) {
366  throw SegmentExpectationViolation("payloads differ. expected \"" + data.value() + "\" but found \"" +
367  std::string(seg.payload().str()) + "\"");
368  }
369  }

372 class TCPSenderTestHarness {
373  std::deque<TCPSegment> outbound_segments;
374  TCPSender sender;
375  std::vector<std::string> steps_executed;
376  std::string name;
377 
378  void collect_output() {
379  while (not sender.segments_out().empty()) {
380  outbound_segments.push_back(std::move(sender.segments_out().front()));
381  sender.segments_out().pop();
382  }
383  }
384 
385  public:
386  TCPSenderTestHarness(const std::string &name_, TCPConfig config)
387  : outbound_segments()
388  , sender(config.send_capacity, config.rt_timeout, config.fixed_isn)
389  , steps_executed()
390  , name(name_) {
391  sender.send_empty_segment();
392  collect_output();
393  std::ostringstream ss;
394  ss << "Initialized with ("
395  << "retx-timeout=" << config.rt_timeout << ")";
396  steps_executed.emplace_back(ss.str());
397  }
398 
399  void execute(const SenderTestStep &step) {
400  try {
401  step.execute(sender, outbound_segments);
402  collect_output();
403  steps_executed.emplace_back(step);
404  } catch (const SenderExpectationViolation &e) {
405  std::cerr << "Test Failure on expectation:\n\t" << std::string(step);
406  std::cerr << "\n\nFailure message:\n\t" << e.what();
407  std::cerr << "\n\nList of steps that executed successfully:";
408  for (const std::string &s : steps_executed) {
409  std::cerr << "\n\t" << s;
410  }
411  std::cerr << std::endl << std::endl;
412  throw SenderExpectationViolation("The test \"" + name + "\" failed");
413  } catch (const std::exception &e) {
414  std::cerr << "Test Failure on expectation:\n\t" << std::string(step);
415  std::cerr << "\n\nFailure message:\n\t" << e.what();
416  std::cerr << "\n\nList of steps that executed successfully:";
417  for (const std::string &s : steps_executed) {
418  std::cerr << "\n\t" << s;
419  }
420  std::cerr << std::endl << std::endl;
421  throw SenderExpectationViolation("The test \"" + name +
422  "\" caused your implementation to throw an exception!");
423  }
424  }
