222  virtual TCPSegment expect_seg(TCPTestHarness &harness) const {
223  if (not harness.can_read()) {
224  throw SegmentExpectationViolation::violated_verb("existed");
225  }
226  TCPSegment seg;
227  if (ParseResult::NoError != seg.parse(harness._flt.read())) {
228  throw SegmentExpectationViolation::violated_verb("was parsable");
229  }
230  if (ack.has_value() and seg.header().ack != ack.value()) {
231  throw SegmentExpectationViolation::violated_field("ack", ack.value(), seg.header().ack);
232  }
233  if (rst.has_value() and seg.header().rst != rst.value()) {
234  throw SegmentExpectationViolation::violated_field("rst", rst.value(), seg.header().rst);
235  }
236  if (syn.has_value() and seg.header().syn != syn.value()) {
237  throw SegmentExpectationViolation::violated_field("syn", syn.value(), seg.header().syn);
238  }
239  if (fin.has_value() and seg.header().fin != fin.value()) {
240  throw SegmentExpectationViolation::violated_field("fin", fin.value(), seg.header().fin);
241  }
242  if (seqno.has_value() and seg.header().seqno != seqno.value()) {
243  throw SegmentExpectationViolation::violated_field("seqno", seqno.value(), seg.header().seqno);
244  }
245  if (ackno.has_value() and seg.header().ackno != ackno.value()) {
246  throw SegmentExpectationViolation::violated_field("ackno", ackno.value(), seg.header().ackno);
247  }
248  if (win.has_value() and seg.header().win != win.value()) {
249  throw SegmentExpectationViolation::violated_field("win", win.value(), seg.header().win);
250  }
251  if (payload_size.has_value() and seg.payload().size() != payload_size.value()) {
252  throw SegmentExpectationViolation::violated_field(
253  "payload_size", payload_size.value(), seg.payload().size());
254  }
255  if (seg.length_in_sequence_space() > TCPConfig::MAX_PAYLOAD_SIZE) {
256  throw SegmentExpectationViolation("packet has length_including_flags (" +
257  std::to_string(seg.length_in_sequence_space()) +
258  ") greater than the maximum");
259  }
260  if (data.has_value() and seg.payload().str() != *data) {
261  throw SegmentExpectationViolation("payloads differ");
262  }
263  return seg;
264  }
