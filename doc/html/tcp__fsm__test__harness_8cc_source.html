144 void TCPTestHarness::execute(const TCPTestStep &step, std::string note) {
145  try {
146  step.execute(*this);
147  while (not _fsm.segments_out().empty()) {
148  _flt.write(_fsm.segments_out().front());
149  _fsm.segments_out().pop();
150  }
151  _steps_executed.emplace_back(step.to_string());
152  } catch (const TCPExpectationViolation &e) {
153  cerr << "Test Failure on expectation:\n\t" << step.to_string();
154  cerr << "\n\nFailure message:\n\t" << e.what();
155  cerr << "\n\nList of steps that executed successfully:";
156  for (const string &s : _steps_executed) {
157  cerr << "\n\t" << s;
158  }
159  cerr << endl << endl;
160  if (note.size() > 0) {
161  cerr << "Note:\n\t" << note << endl << endl;
162  }
163  throw e;
164  }
165 }
